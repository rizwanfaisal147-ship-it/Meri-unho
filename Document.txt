  <script src="https://cdn.jsdelivr.net/npm/three@0.128.0/build/three.min.js"></script>
  <script src="https://cdn.jsdelivr.net/npm/gsap@3.12.5/dist/gsap.min.js"></script>
  <script src="https://cdn.jsdelivr.net/npm/gsap@3.12.5/dist/ScrollTrigger.min.js"></script>
  <script>
    (function() {
      // register scrolltrigger
      gsap.registerPlugin(ScrollTrigger);

      // ----- THREE.JS particles (premium dark gold + pink) -----
      const canvas = document.getElementById('bg-canvas');
      const scene = new THREE.Scene();
      scene.background = new THREE.Color(0x0b0a0c);
      const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
      camera.position.z = 30;
      const renderer = new THREE.WebGLRenderer({ canvas, alpha: false, antialias: true });
      renderer.setSize(window.innerWidth, window.innerHeight);
      renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));

      const geometry = new THREE.BufferGeometry();
      const count = 2000;
      const positions = new Float32Array(count * 3);
      const colors = new Float32Array(count * 3);
      for (let i = 0; i < count; i++) {
        let r = (Math.random() - 0.5) * 60;
        let x = (Math.random() - 0.5) * 80;
        let y = (Math.random() - 0.5) * 80;
        let z = (Math.random() - 0.5) * 60;
        positions[i*3] = x; positions[i*3+1] = y; positions[i*3+2] = z;
        // gold / soft pink colors
        let color = new THREE.Color().setHSL(0.08 + Math.random()*0.1, 0.8, 0.6 + Math.random()*0.3);
        colors[i*3] = color.r; colors[i*3+1] = color.g; colors[i*3+2] = color.b;
      }
      geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));
      geometry.setAttribute('color', new THREE.BufferAttribute(colors, 3));
      const material = new THREE.PointsMaterial({ size: 0.2, vertexColors: true, transparent: true, opacity: 0.7, blending: THREE.AdditiveBlending });
      const particles = new THREE.Points(geometry, material);
      scene.add(particles);

      function animateParticles() {
        requestAnimationFrame(animateParticles);
        particles.rotation.y += 0.0001;
        particles.rotation.x += 0.00005;
        renderer.render(scene, camera);
      }
      animateParticles();

      window.addEventListener('resize', () => {
        camera.aspect = window.innerWidth / window.innerHeight;
        camera.updateProjectionMatrix();
        renderer.setSize(window.innerWidth, window.innerHeight);
      });

      // intro logic (music removed)
      const intro = document.getElementById('introScreen');
      const main = document.getElementById('mainContent');
      const enterBtn = document.getElementById('enterBtn');

      enterBtn.addEventListener('click', () => {
        gsap.to(intro, { opacity: 0, duration: 1.5, ease: 'power2.inOut', onComplete: () => {
          intro.style.display = 'none';
          main.style.opacity = 1;
        }});
      });

      // typewriter effect
      const message = `I remember the first time we spoke — the world felt lighter. 
Every moment with you is a page from a fairytale I never want to end. 
You have a way of making ordinary days feel extraordinary. 
And today, the universe celebrates the masterpiece that is you. ??`;

      const container = document.getElementById('typewriterContainer');
      let i = 0;
      function typeNext() {
        if (i < message.length) {
          container.innerHTML += message.charAt(i) === '\n' ? '<br/>' : message.charAt(i);
          i++;
          setTimeout(typeNext, 50);
        } else {
          // add glowing words after
          container.innerHTML = container.innerHTML.replace('fairytale', '<span class="glow-word">fairytale</span>');
          container.innerHTML = container.innerHTML.replace('masterpiece', '<span class="glow-word">masterpiece</span>');
        }
      }
      // start typewriter when in view (approx)
      gsap.timeline({
        scrollTrigger: {
          trigger: '#typewriterContainer',
          start: 'top 80%',
          onEnter: () => { if(i===0) typeNext(); }
        }
      });

      // sparkles hero
      setInterval(() => {
        let heart = document.createElement('span');
        heart.className = 'floating-heart';
        heart.innerHTML = ['?','??','??','??','??'][Math.floor(Math.random()*5)];
        heart.style.left = Math.random() * 100 + '%';
        heart.style.top = Math.random() * 100 + '%';
        heart.style.animation = `float ${3+Math.random()*4}s linear`;
        heart.style.opacity = 0.8;
        document.getElementById('hero').appendChild(heart);
        setTimeout(() => heart.remove(), 4000);
      }, 400);

      // interactive cake
      const cake = document.getElementById('cakeContainer');
      const wishMsg = document.getElementById('wishMessage');
      cake.addEventListener('click', () => {
        cake.classList.add('blow-out');
        cake.innerHTML = '??<span style="font-size:4rem;">??</span>';
        wishMsg.style.opacity = 1;
        // trigger fireworks (soft confetti)
        for (let i=0;i<12;i++) {
          let f = document.createElement('div');
          f.style.position = 'fixed'; f.style.left='50%'; f.style.top='50%';
          f.style.color = '#e5c687'; f.style.fontSize='2rem'; f.style.pointerEvents='none';
          f.innerHTML = ['?','??','??','??'][Math.floor(Math.random()*4)];
          f.style.zIndex = 9999;
          document.body.appendChild(f);
          gsap.to(f, { x: (Math.random()-0.5)*400, y: (Math.random()-0.5)*400, opacity:0, duration:2, ease:'power2.out', onComplete:()=>f.remove() });
        }
      });

      // secret modal
      const secretTrigger = document.getElementById('secretTrigger');
      const secretModal = document.getElementById('secretModal');
      const closeSecret = document.getElementById('closeSecret');
      secretTrigger.addEventListener('click', () => secretModal.classList.add('show'));
      closeSecret.addEventListener('click', () => secretModal.classList.remove('show'));
      secretModal.addEventListener('click', (e) => { if(e.target === secretModal) secretModal.classList.remove('show'); });

      // final wishes soft fade-out effect? (we just add a subtle entrance)
      gsap.from('#finalWishes .glass-card', { scrollTrigger: '#finalWishes', scale: 0.8, opacity: 0, duration: 1.5 });

      // floating sparkles on final section
      setInterval(() => {
        let sp = document.createElement('span');
        sp.innerHTML = '?'; sp.style.position='absolute'; sp.style.left=Math.random()*100+'%'; sp.style.top=Math.random()*100+'%';
        sp.style.fontSize='2rem'; sp.style.opacity=0.6; sp.style.pointerEvents='none';
        sp.style.animation = 'float 5s infinite';
        document.getElementById('finalWishes').appendChild(sp);
        setTimeout(() => sp.remove(), 5000);
      }, 500);

      // add keyframe style
      const styleSheet = document.createElement("style");
      styleSheet.textContent = `@keyframes float { 0% { transform: translateY(0) rotate(0deg); opacity:0.6; } 100% { transform: translateY(-120px) rotate(20deg); opacity:0; } }`;
      document.head.appendChild(styleSheet);

      // editable wish (double-click to edit)
      const wishP = document.getElementById('editableWish');
      wishP.addEventListener('dblclick', () => {
        wishP.contentEditable = true;
        wishP.focus();
      });
      wishP.addEventListener('blur', () => wishP.contentEditable = false);
    })();
  </script>
